```python
def embedding_plot(weights):

  pca = PCA(n_components=2)
  weights = pca.fit_transform(weights)

  embedding_df = pd.DataFrame({
      'word' : vocab, 
      'x0'   : weights[:,0],
      'x1'   : weights[:,1]
  })

  fig = px.scatter(embedding_df, 
                 x = "x0", 
                 y = "x1", 
                 size = list(np.ones(len(embedding_df))),
                 size_max = 2,
                 hover_name = "word")
  return fig
```
